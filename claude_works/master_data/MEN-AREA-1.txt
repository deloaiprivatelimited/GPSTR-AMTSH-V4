[ಮೂಲ_ಅವಧಾರಣೆ_1]
concept_id: MEN-AREA-1_concept_1
concept_name: ಬಹುಭುಜಾಕೃತಿಗಳ ಸುತ್ತಳತೆ (Perimeter of Polygons)
domain: Mensuration

  [ಸಾರಾಂಶ]
  ಯಾವುದೇ ಆವೃತ ಸಮತಲಾಕೃತಿಯ ಗಡಿಯ ಒಟ್ಟು ಉದ್ದವನ್ನು ಅದರ ಸುತ್ತಳತೆ ಅಥವಾ ಪರಿಧಿ ಎಂದು ಕರೆಯಲಾಗುತ್ತದೆ. ಬಹುಭುಜಾಕೃತಿಗಳ ಸಂದರ್ಭದಲ್ಲಿ, ಇದು ಅದರ ಎಲ್ಲಾ ಬಾಹುಗಳ ಉದ್ದಗಳ ಮೊತ್ತವಾಗಿರುತ್ತದೆ. ಆಯತ, ಚೌಕ, ಮತ್ತು ತ್ರಿಭುಜದಂತಹ ಸಾಮಾನ್ಯ ಆಕೃತಿಗಳಿಗೆ ನಿರ್ದಿಷ್ಟ ಸೂತ್ರಗಳನ್ನು ಬಳಸಿ ಸುತ್ತಳತೆಯನ್ನು ಸುಲಭವಾಗಿ ಲೆಕ್ಕಾಚಾರ ಮಾಡಬಹುದು.

  [ವ್ಯಾಖ್ಯೆ]
  - **ಸುತ್ತಳತೆ (Perimeter):** ಯಾವುದೇ ಆವೃತ ಸಮತಲಾಕೃತಿಯ ಗಡಿಯ ಸುತ್ತಲೂ ಒಂದು ಬಾರಿ ಚಲಿಸಿದ ಒಟ್ಟು ದೂರವು ಅದರ ಸುತ್ತಳತೆಯಾಗಿರುತ್ತದೆ.
  - **ಬಹುಭುಜಾಕೃತಿಯ ಸುತ್ತಳತೆ (Perimeter of a Polygon):** ಒಂದು ಬಹುಭುಜಾಕೃತಿಯ ಸುತ್ತಳತೆಯು ಅದರ ಎಲ್ಲಾ ಬಾಹುಗಳ ಉದ್ದಗಳ ಮೊತ್ತವಾಗಿರುತ್ತದೆ.
  - **ನಿಯಮಿತ ಬಹುಭುಜಾಕೃತಿ (Regular Polygon):** ಎಲ್ಲಾ ಬಾಹುಗಳು ಮತ್ತು ಎಲ್ಲಾ ಕೋನಗಳು ಸಮವಿರುವ ಆವೃತ ಆಕೃತಿಗಳನ್ನು ನಿಯಮಿತ ಬಹುಭುಜಾಕೃತಿಗಳು ಎಂದು ಕರೆಯುತ್ತೇವೆ.

  [ಸೂತ್ರಗಳು]
  - formula_name: "Perimeter of a Polygon"
    latex: "P = \\text{sum of the lengths of all sides}"
    condition: "ಯಾವುದೇ ಬಹುಭುಜಾಕೃತಿಗೆ ಅನ್ವಯಿಸುತ್ತದೆ."
    used_in: "ಯಾವುದೇ ಬಹುಭುಜಾಕೃತಿಯ ಸುತ್ತಳತೆಯನ್ನು ಕಂಡುಹಿಡಿಯಲು."

  - formula_name: "Perimeter of a Rectangle"
    latex: "P = 2(l+b)"
    condition: "l = ಆಯತದ ಉದ್ದ (length), b = ಆಯತದ ಅಗಲ (breadth)."
    derived_from: "P = l + b + l + b"
    used_in: "ಆಯತಾಕಾರದ ವಸ್ತುಗಳ ಸುತ್ತಳತೆ ಲೆಕ್ಕಾಚಾರ ಮಾಡಲು."

  - formula_name: "Perimeter of a Square"
    latex: "P = 4s"
    condition: "s = ಚೌಕದ ಒಂದು ಬಾಹುವಿನ ಉದ್ದ (side)."
    derived_from: "P = s + s + s + s"
    used_in: "ಚೌಕಾಕಾರದ ವಸ್ತುಗಳ ಸುತ್ತಳತೆ ಲೆಕ್ಕಾಚಾರ ಮಾಡಲು."

  - formula_name: "Perimeter of a Triangle"
    latex: "P = a+b+c"
    condition: "a, b, c ಗಳು ತ್ರಿಭುಜದ ಮೂರು ಬಾಹುಗಳ ಉದ್ದಗಳು."
    used_in: "ಯಾವುದೇ ತ್ರಿಭುಜದ ಸುತ್ತಳತೆ ಲೆಕ್ಕಾಚಾರ ಮಾಡಲು."

  - formula_name: "Perimeter of an Equilateral Triangle"
    latex: "P = 3s"
    condition: "s = ಸಮಬಾಹು ತ್ರಿಭುಜದ ಒಂದು ಬಾಹುವಿನ ಉದ್ದ."
    derived_from: "P = s + s + s"
    used_in: "ಸಮಬಾಹು ತ್ರಿಭುಜದ ಸುತ್ತಳತೆ ಲೆಕ್ಕಾಚಾರ ಮಾಡಲು."

  [ಕ್ರಮಬದ್ಧ ಉದಾಹರಣೆ]
  - example_id: "MEN-AREA-1_example_1"
    difficulty: "basic"
    problem: "3 m ಉದ್ದ ಮತ್ತು 2 m ಅಗಲವಿರುವ ಆಯತಾಕಾರದ ಮೇಜಿನ ಬಟ್ಟೆಯ ಸುತ್ತಲೂ ಲೇಸ್ ಹಾಕಲು ಅಕ್ಷಿ ಬಯಸುತ್ತಾಳೆ. ಅಗತ್ಯವಿರುವ ಲೇಸ್‌ನ ಉದ್ದ ಕಂಡುಹಿಡಿಯಿರಿ."
    steps:
      - step: 1
        action: "ದತ್ತಾಂಶಗಳನ್ನು ಗುರುತಿಸುವುದು."
        justification: "ಸಮಸ್ಯೆಯಲ್ಲಿ ಆಯತದ ಉದ್ದ ಮತ್ತು ಅಗಲವನ್ನು ನೀಡಲಾಗಿದೆ."
        calculation: "ಉದ್ದ (l) = 3 m, ಅಗಲ (b) = 2 m."
      - step: 2
        action: "ಆಯತದ ಸುತ್ತಳತೆಯ ಸೂತ್ರವನ್ನು ಬಳಸುವುದು."
        justification: "ಅಗತ್ಯವಿರುವ ಲೇಸ್‌ನ ಉದ್ದವು ಮೇಜಿನ ಬಟ್ಟೆಯ ಸುತ್ತಳತೆಗೆ ಸಮವಾಗಿರುತ್ತದೆ."
        calculation: "ಸುತ್ತಳತೆ = 2 × (ಉದ್ದ + ಅಗಲ)"
      - step: 3
        action: "ಸೂತ್ರದಲ್ಲಿ ಬೆಲೆಗಳನ್ನು ಆದೇಶಿಸುವುದು."
        justification: "ಉದ್ದ ಮತ್ತು ಅಗಲದ ಮೌಲ್ಯಗಳನ್ನು ಸೂತ್ರದಲ್ಲಿ ಹಾಕಿ ಲೆಕ್ಕಾಚಾರ ಮಾಡುವುದು."
        calculation: "ಸುತ್ತಳತೆ = 2 × (3 m + 2 m) = 2 × 5 m = 10 m."
    result: "ಅಗತ್ಯವಿರುವ ಲೇಸ್‌ನ ಉದ್ದ 10 ಮೀಟರ್."

    steps:
      - step: 1
        action: "ಚೌಕಾಕಾರದ ಉದ್ಯಾನವನದ ಸುತ್ತಳತೆಯನ್ನು ಕಂಡುಹಿಡಿಯುವುದು."
        justification: "ಒಂದು ಸುತ್ತಿನಲ್ಲಿ ಕ್ರಮಿಸಿದ ದೂರವು ಉದ್ಯಾನವನದ ಸುತ್ತಳತೆಗೆ ಸಮ."
        calculation: "ಚೌಕದ ಸುತ್ತಳತೆ = 4 × ಒಂದು ಬಾಹುವಿನ ಉದ್ದ = 4 × 75 m = 300 m."
      - step: 2
        action: "ಒಂದು ಸುತ್ತಿನಲ್ಲಿ ಕ್ರಮಿಸಿದ ದೂರವನ್ನು ಗುರುತಿಸುವುದು."
        justification: "ಸುತ್ತಳತೆಯೇ ಒಂದು ಸುತ್ತಿನ ದೂರ."
        calculation: "ಒಂದು ಸುತ್ತಿನಲ್ಲಿ ಕ್ರಮಿಸಿದ ದೂರ = 300 m."
      - step: 3
        action: "ಮೂರು ಸುತ್ತುಗಳಲ್ಲಿ ಕ್ರಮಿಸಿದ ಒಟ್ಟು ದೂರವನ್ನು ಲೆಕ್ಕಾಚಾರ ಮಾಡುವುದು."
        justification: "ಒಟ್ಟು ದೂರ = ಸುತ್ತುಗಳ ಸಂಖ್ಯೆ × ಒಂದು ಸುತ್ತಿನ ದೂರ."

    result: "ಉಷಾ ಮೂರು ಸುತ್ತುಗಳಲ್ಲಿ ಕ್ರಮಿಸಿದ ಒಟ್ಟು ದೂರ 900 ಮೀಟರ್."

  [ವಿಶ್ಲೇಷಣೆ]
  ಸುತ್ತಳತೆಯು ಒಂದು ರೇಖೀಯ ಅಳತೆಯಾಗಿದ್ದು (ಉದಾ. cm, m), ಇದು ಒಂದು ಆಕೃತಿಯ ಗಡಿಯ ಉದ್ದವನ್ನು ಪ್ರತಿನಿಧಿಸುತ್ತದೆ. ವಿಸ್ತೀರ್ಣದೊಂದಿಗೆ ಇದನ್ನು ಗೊಂದಲ ಮಾಡಿಕೊಳ್ಳಬಾರದು, ವಿಸ್ತೀರ್ಣವು ಆಕೃತಿಯು ಆವರಿಸುವ ಪ್ರದೇಶವನ್ನು ಅಳೆಯುತ್ತದೆ. ಸ್ಪರ್ಧಾತ್ಮಕ ಪರೀಕ್ಷೆಗಳಲ್ಲಿ, ಸುತ್ತಳತೆಯ ಸೂತ್ರಗಳನ್ನು ನೇರವಾಗಿ ಬಳಸುವ ಅಥವಾ ಬೇಲಿ ಹಾಕುವ ವೆಚ್ಚ, ಟ್ರ್ಯಾಕ್‌ನಲ್ಲಿ ಓಡಿದ ದೂರದಂತಹ ಅನ್ವಯಿಕ ಸಮಸ್ಯೆಗಳನ್ನು ಕೇಳಲಾಗುತ್ತದೆ.

  [ಸಂಬಂಧ]
  - ಪೂರ್ವಾಪೇಕ್ಷಿತ ಅವಧಾರಣೆಗಳು: ಉದ್ದದ ಮೂಲಭೂತ ಪರಿಕಲ್ಪನೆ, ಸಂಕಲನ.
  - ಮುಂದಿನ ಅವಧಾರಣೆಗಳೊಂದಿಗೆ ಸಂಪರ್ಕ: ವಿಸ್ತೀರ್ಣ, ಸಂಯುಕ್ತ ಆಕೃತಿಗಳು, ವೃತ್ತದ ಪರಿಧಿ.

  - 2-ಅಂಕ ಪ್ರಶ್ನೆ: ಬೇಲಿ ಹಾಕುವ ವೆಚ್ಚ ಅಥವಾ ಬಹುಭುಜಾಕೃತಿಯ ಸುತ್ತಳತೆ ಕಂಡುಹಿಡಿಯುವ ಸರಳ ಲೆಕ್ಕಗಳು.
  - 5-ಅಂಕ ಪ್ರಶ್ನೆ: ಅನ್ವಯಿಕ ಸಮಸ್ಯೆಗಳು, ಉದಾಹರಣೆಗೆ ಒಂದು ಆಕಾರದ ತಂತಿಯನ್ನು ಮತ್ತೊಂದು ಆಕಾರಕ್ಕೆ ಪರಿವರ್ತಿಸಿದಾಗ ಆಗುವ ಬದಲಾವಣೆಗಳು.
  - ಪ್ರಮುಖ trap: ಸುತ್ತಳತೆ ಮತ್ತು ವಿಸ್ತೀರ್ಣದ ಘಟಕಗಳನ್ನು (units) ಅದಲು ಬದಲು ಮಾಡುವುದು. (ಉದಾ. cm vs cm²).

[ಮೂಲ_ಅವಧಾರಣೆ_2]
concept_id: MEN-AREA-1_concept_2
concept_name: ಬಹುಭುಜಾಕೃತಿಗಳ ವಿಸ್ತೀರ್ಣ (Area of Polygons)
domain: Mensuration

  [ಸಾರಾಂಶ]
  ವಿಸ್ತೀರ್ಣವು ಒಂದು ಆವೃತ ಆಕೃತಿಯು ಸಮತಲದಲ್ಲಿ ಆವರಿಸುವ ಪ್ರದೇಶದ ಪ್ರಮಾಣವನ್ನು ಅಳೆಯುತ್ತದೆ. ಇದನ್ನು ಚದರ ಮಾನಗಳಲ್ಲಿ (ಉದಾ. cm², m²) ಅಳೆಯಲಾಗುತ್ತದೆ. ಆಯತ ಮತ್ತು ಚೌಕಗಳ ವಿಸ್ತೀರ್ಣವನ್ನು ಅವುಗಳ ಬಾಹುಗಳ ಉದ್ದವನ್ನು ಬಳಸಿ ನಿರ್ದಿಷ್ಟ ಸೂತ್ರಗಳಿಂದ ಕಂಡುಹಿಡಿಯಬಹುದು.

  [ವ್ಯಾಖ್ಯೆ]
  - **ವಿಸ್ತೀರ್ಣ (Area):** ಒಂದು ಆವೃತ ಆಕೃತಿಯಿಂದ ಆವರಿಸಿರುವ ಪ್ರದೇಶದ ಪ್ರಮಾಣವನ್ನು ಅದರ ವಿಸ್ತೀರ್ಣ ಎಂದು ಕರೆಯುತ್ತೇವೆ.

  [ಸೂತ್ರಗಳು]
  - formula_name: "Area of a Rectangle"
    latex: "A = l \\times b"
    condition: "l = ಆಯತದ ಉದ್ದ (length), b = ಆಯತದ ಅಗಲ (breadth)."
    used_in: "ಆಯತಾಕಾರದ ಪ್ರದೇಶಗಳ ವಿಸ್ತೀರ್ಣವನ್ನು ಲೆಕ್ಕಾಚಾರ ಮಾಡಲು."

  - formula_name: "Area of a Square"
    latex: "A = s^2"
    condition: "s = ಚೌಕದ ಒಂದು ಬಾಹುವಿನ ಉದ್ದ (side)."
    derived_from: "A = s × s"
    used_in: "ಚೌಕಾಕಾರದ ಪ್ರದೇಶಗಳ ವಿಸ್ತೀರ್ಣವನ್ನು ಲೆಕ್ಕಾಚಾರ ಮಾಡಲು."

  [ಕ್ರಮಬದ್ಧ ಉದಾಹರಣೆ]
  - example_id: "MEN-AREA-1_example_3"
    difficulty: "intermediate"
    problem: "ಒಂದು ನೆಲದ ಉದ್ದ 5 m ಮತ್ತು ಅಗಲ 4 m ಇದೆ. ಬಾಹುವಿನ ಉದ್ದ 3 m ಇರುವ ಒಂದು ಚೌಕಾಕಾರದ ಕಾರ್ಪೆಟ್‌ನ್ನು ನೆಲದ ಮೇಲೆ ಹಾಸಿದೆ. ಕಾರ್ಪೆಟ್ ಹಾಕದಿರುವ ನೆಲದ ಭಾಗದ ವಿಸ್ತೀರ್ಣವನ್ನು ಕಂಡುಹಿಡಿಯಿರಿ."
    steps:
      - step: 1
        action: "ನೆಲದ ಒಟ್ಟು ವಿಸ್ತೀರ್ಣವನ್ನು ಲೆಕ್ಕಾಚಾರ ಮಾಡುವುದು."
        justification: "ನೆಲವು ಆಯತಾಕಾರದಲ್ಲಿದೆ, ಆದ್ದರಿಂದ ಆಯತದ ವಿಸ್ತೀರ್ಣದ ಸೂತ್ರವನ್ನು ಬಳಸಬೇಕು."
        calculation: "ನೆಲದ ವಿಸ್ತೀರ್ಣ = ಉದ್ದ × ಅಗಲ = 5 m × 4 m = 20 ಚದರ ಮೀಟರ್."
      - step: 2
        action: "ಕಾರ್ಪೆಟ್‌ನ ವಿಸ್ತೀರ್ಣವನ್ನು ಲೆಕ್ಕಾಚಾರ ಮಾಡುವುದು."
        justification: "ಕಾರ್ಪೆಟ್ ಚೌಕಾಕಾರದಲ್ಲಿದೆ, ಆದ್ದರಿಂದ ಚೌಕದ ವಿಸ್ತೀರ್ಣದ ಸೂತ್ರವನ್ನು ಬಳಸಬೇಕು."
        calculation: "ಕಾರ್ಪೆಟ್‌ನ ವಿಸ್ತೀರ್ಣ = ಬಾಹು × ಬಾಹು = 3 m × 3 m = 9 ಚದರ ಮೀಟರ್."
      - step: 3
        action: "ಕಾರ್ಪೆಟ್ ಹಾಕದಿರುವ ಭಾಗದ ವಿಸ್ತೀರ್ಣವನ್ನು ಕಂಡುಹಿಡಿಯುವುದು."
        justification: "ಉಳಿದ ವಿಸ್ತೀರ್ಣ = ಒಟ್ಟು ವಿಸ್ತೀರ್ಣ - ಕಾರ್ಪೆಟ್‌ನ ವಿಸ್ತೀರ್ಣ."
        calculation: "ಉಳಿದ ವಿಸ್ತೀರ್ಣ = 20 m² - 9 m² = 11 ಚದರ ಮೀಟರ್."
    result: "ಕಾರ್ಪೆಟ್ ಹಾಕದಿರುವ ನೆಲದ ಭಾಗದ ವಿಸ್ತೀರ್ಣ 11 ಚದರ ಮೀಟರ್."

  [ವಿಶ್ಲೇಷಣೆ]
  ವಿಸ್ತೀರ್ಣದ ಪರಿಕಲ್ಪನೆಯು ಜಾಗವನ್ನು ಅಳೆಯಲು ಮತ್ತು ಹೋಲಿಸಲು ಅತ್ಯಗತ್ಯ. ಸ್ಪರ್ಧಾತ್ಮಕ ಪರೀಕ್ಷೆಗಳಲ್ಲಿ, ನೆಲಹಾಸು ಹಾಕುವ, ಗೋಡೆಗೆ ಬಣ್ಣ ಬಳಿಯುವ, ಅಥವಾ ಕೃಷಿ ಭೂಮಿಯಂತಹ ಅನ್ವಯಿಕ ಸಮಸ್ಯೆಗಳಲ್ಲಿ ವಿಸ್ತೀರ್ಣದ ಲೆಕ್ಕಾಚಾರಗಳು ಬರುತ್ತವೆ. ಸಂಯುಕ್ತ ಆಕೃತಿಗಳ ವಿಸ್ತೀರ್ಣವನ್ನು ಕಂಡುಹಿಡಿಯಲು, ಅವುಗಳನ್ನು ಸರಳ ಆಯತಗಳು ಮತ್ತು ಚೌಕಗಳಾಗಿ ವಿಭಜಿಸುವ ತಂತ್ರವು ಮುಖ್ಯವಾಗಿದೆ.

  [ಸಂಬಂಧ]
  - ಪೂರ್ವಾಪೇಕ್ಷಿತ ಅವಧಾರಣೆಗಳು: ಸುತ್ತಳತೆ, ಗುಣಾಕಾರ.
  - ಮುಂದಿನ ಅವಧಾರಣೆಗಳೊಂದಿಗೆ ಸಂಪರ್ಕ: ಸಮಾಂತರ ಚತುರ್ಭುಜ ಮತ್ತು ತ್ರಿಭುಜದ ವಿಸ್ತೀರ್ಣ, ವೃತ್ತದ ವಿಸ್ತೀರ್ಣ, ಘನಫಲ (Volume).

  [ಶೈಕ್ಷಣಿಕ_ದೃಷ್ಟಿಕೋನ]
  - gpstr_weightage: high
  - MCQ ಸಾಧ್ಯತೆ: ವಿಸ್ತೀರ್ಣ ನೀಡಿ ಬಾಹುವನ್ನು ಕಂಡುಹಿಡಿಯುವುದು, ಘಟಕ ಪರಿವರ್ತನೆ (unit conversion).
  - 2-ಅಂಕ ಪ್ರಶ್ನೆ: ನೇರ ಸೂತ್ರದ ಬಳಕೆ ಅಥವಾ ಮೇಲಿನ ಉದಾಹರಣೆಯಂತಹ ಸರಳ ಅನ್ವಯಿಕ ಸಮಸ್ಯೆ.
  - 5-ಅಂಕ ಪ್ರಶ್ನೆ: ಸಂಯುಕ್ತ ಆಕೃತಿಗಳ ವಿಸ್ತೀರ್ಣ, ಅಥವಾ ವೆಚ್ಚವನ್ನು ಕಂಡುಹಿಡಿಯುವ ಬಹು-ಹಂತದ ಸಮಸ್ಯೆಗಳು.
  - ಪ್ರಮುಖ trap: ಸುತ್ತಳತೆ ಮತ್ತು ವಿಸ್ತೀರ್ಣವನ್ನು ಗೊಂದಲ ಮಾಡಿಕೊಳ್ಳುವುದು. ಉದ್ದ ಮತ್ತು ಅಗಲವನ್ನು ಸರಿಯಾಗಿ ಗುರುತಿಸದಿರುವುದು.

[ಮೂಲ_ಅವಧಾರಣೆ_3]
concept_id: MEN-AREA-1_concept_3
concept_name: ಸಮಾಂತರ ಚತುರ್ಭುಜದ ವಿಸ್ತೀರ್ಣ (Area of a Parallelogram)
domain: Mensuration

  [ಸಾರಾಂಶ]
  ಸಮಾಂತರ ಚತುರ್ಭುಜದ ವಿಸ್ತೀರ್ಣವನ್ನು ಅದರ ಪಾದ ಮತ್ತು ಆ ಪಾದಕ್ಕೆ ಎಳೆದ ಅನುರೂಪ ಎತ್ತರವನ್ನು ಗುಣಿಸುವ ಮೂಲಕ ಕಂಡುಹಿಡಿಯಲಾಗುತ್ತದೆ. ಒಂದು ಸಮಾಂತರ ಚತುರ್ಭುಜವನ್ನು ಕತ್ತರಿಸಿ ಪುನರ್ಜೋಡಿಸುವ ಮೂಲಕ, ಅದೇ ಪಾದ ಮತ್ತು ಎತ್ತರವನ್ನು ಹೊಂದಿರುವ ಸಮವಿಸ್ತೀರ್ಣದ ಆಯತವಾಗಿ ಪರಿವರ್ತಿಸಬಹುದು.

  [ಸೂತ್ರಗಳು]
  - formula_name: "Area of a Parallelogram"
    latex: "A = b \\times h"
    condition: "b = ಸಮಾಂತರ ಚತುರ್ಭುಜದ ಪಾದ (base), h = ಪಾದಕ್ಕೆ ಎಳೆದ ಅನುರೂಪ ಎತ್ತರ (corresponding height)."
    used_in: "ಸಮಾಂತರ ಚತುರ್ಭುಜದ ವಿಸ್ತೀರ್ಣವನ್ನು ಲೆಕ್ಕಾಚಾರ ಮಾಡಲು."
    derivation:
      - step: 1
        action: "ಒಂದು ಸಮಾಂತರ ಚತುರ್ಭುಜವನ್ನು ಗ್ರಾಫ್ ಹಾಳೆಯ ಮೇಲೆ ರಚಿಸಿ."
        justification: "ವಿಸ್ತೀರ್ಣವನ್ನು ದೃಶ್ಯೀಕರಿಸಲು."
      - step: 2
        action: "ಒಂದು ಶೃಂಗದಿಂದ ಅಭಿಮುಖ ಬಾಹುವಿಗೆ ಲಂಬವನ್ನು ಎಳೆದು, ಒಂದು ಲಂಬಕೋನ ತ್ರಿಭುಜವನ್ನು ಕತ್ತರಿಸಿ."
        justification: "ಆಕೃತಿಯನ್ನು ಮರುರೂಪಿಸಲು."
      - step: 3
        action: "ಕತ್ತರಿಸಿದ ತ್ರಿಭುಜವನ್ನು ಸಮಾಂತರ ಚತುರ್ಭುಜದ ಇನ್ನೊಂದು ಬದಿಗೆ ಜೋಡಿಸಿ."
        justification: "ಹೊಸ ಆಕೃತಿಯನ್ನು ರಚಿಸಲು."
      - step: 4
        action: "ಉಂಟಾದ ಆಕೃತಿಯು ಒಂದು ಆಯತ ಎಂದು ಗಮನಿಸಿ."
        justification: "ಹೊಸ ಆಕೃತಿಯ ಗುಣಲಕ್ಷಣಗಳನ್ನು ಪರಿಶೀಲಿಸುವುದು."
      - step: 5
        action: "ಆಯತದ ವಿಸ್ತೀರ್ಣವನ್ನು ಸಮಾಂತರ ಚತುರ್ಭುಜದ ವಿಸ್ತೀರ್ಣಕ್ಕೆ ಸಮೀಕರಿಸಿ."
        justification: "ವಿಸ್ತೀರ್ಣವು ಬದಲಾಗಿಲ್ಲ (Area is conserved)."
        calculation: "ಸಮಾಂತರ ಚತುರ್ಭುಜದ ವಿಸ್ತೀರ್ಣ = ಆಯತದ ವಿಸ್ತೀರ್ಣ = ಉದ್ದ × ಅಗಲ."
      - step: 6
        action: "ಆಯತದ ಉದ್ದ ಮತ್ತು ಅಗಲವನ್ನು ಸಮಾಂತರ ಚತುರ್ಭುಜದ ಪಾದ ಮತ್ತು ಎತ್ತರಕ್ಕೆ ಸಂಬಂಧಿಸಿ."
        justification: "ಪರಿವರ್ತನೆಯಿಂದ, ಆಯತದ ಉದ್ದವು ಸಮಾಂತರ ಚತುರ್ಭುಜದ ಪಾದಕ್ಕೂ (b) ಮತ್ತು ಆಯತದ ಅಗಲವು ಸಮಾಂತರ ಚತುರ್ಭುಜದ ಎತ್ತರಕ್ಕೂ (h) ಸಮನಾಗಿರುತ್ತದೆ."
        calculation: "ವಿಸ್ತೀರ್ಣ = b × h."
    result: "ಸಮಾಂತರ ಚತುರ್ಭುಜದ ವಿಸ್ತೀರ್ಣ = ಪಾದ × ಎತ್ತರ."

  [ಅಂಶಪಟ್ಟಿ / Properties]
  - ಸಮಾಂತರ ಚತುರ್ಭುಜದ ಯಾವುದೇ ಬಾಹುವನ್ನು ಅದರ ಪಾದವಾಗಿ ಪರಿಗಣಿಸಬಹುದು.
  - ಎತ್ತರವು ಯಾವಾಗಲೂ ಆಯ್ದ ಪಾದಕ್ಕೆ ಲಂಬವಾಗಿರಬೇಕು.
  - ಒಂದೇ ಪಾದ ಮತ್ತು ಒಂದೇ ಎತ್ತರವನ್ನು ಹೊಂದಿರುವ ಎಲ್ಲಾ ಸಮಾಂತರ ಚತುರ್ಭುಜಗಳು ಸಮಾನ ವಿಸ್ತೀರ್ಣವನ್ನು ಹೊಂದಿರುತ್ತವೆ.

  [ಕ್ರಮಬದ್ಧ ಉದಾಹರಣೆ]
  - example_id: "MEN-AREA-1_example_4"
    difficulty: "intermediate"
    problem: "ಸಮಾಂತರ ಚತುರ್ಭುಜ ABCD ಯ ಬಾಹುಗಳು 6 cm ಮತ್ತು 4 cm ಆಗಿದೆ. ಪಾದ CD (6 cm) ಯ ಅನುರೂಪ ಎತ್ತರ 3cm. (i) ಸಮಾಂತರ ಚತುರ್ಭುಜದ ವಿಸ್ತೀರ್ಣ, ಮತ್ತು (ii) ಪಾದ AD (4 cm) ಗೆ ಅನುರೂಪವಾದ ಎತ್ತರವನ್ನು ಕಂಡುಹಿಡಿಯಿರಿ."
    steps:
      - step: 1
        action: "(i) ವಿಸ್ತೀರ್ಣವನ್ನು ಲೆಕ್ಕಾಚಾರ ಮಾಡುವುದು."
        justification: "ಪಾದ CD ಮತ್ತು ಅದರ ಅನುರೂಪ ಎತ್ತರವನ್ನು ನೀಡಲಾಗಿದೆ."
        calculation: "ವಿಸ್ತೀರ್ಣ = ಪಾದ × ಎತ್ತರ = 6 cm × 3 cm = 18 cm²."
      - step: 2
        action: "(ii) ಇನ್ನೊಂದು ಪಾದಕ್ಕೆ ಅನುರೂಪವಾದ ಎತ್ತರವನ್ನು ಕಂಡುಹಿಡಿಯುವುದು."
        justification: "ಸಮಾಂತರ ಚತುರ್ಭುಜದ ವಿಸ್ತೀರ್ಣವು ಯಾವುದೇ ಪಾದ ಮತ್ತು ಅದರ ಅನುರೂಪ ಎತ್ತರವನ್ನು ಬಳಸಿದರೂ ಒಂದೇ ಆಗಿರುತ್ತದೆ. ಈಗ ಪಾದ AD ಯನ್ನು ಬಳಸೋಣ."
        calculation: "ವಿಸ್ತೀರ್ಣ = ಪಾದ AD × ಅನುರೂಪ ಎತ್ತರ (x)"
      - step: 3
        action: "ಸಮೀಕರಣವನ್ನು ಬಿಡಿಸುವುದು."
        justification: "ಹಂತ 1 ರಿಂದ ವಿಸ್ತೀರ್ಣದ ಮೌಲ್ಯವನ್ನು ಬಳಸಿ."
        calculation: "18 cm² = 4 cm × x \n x = 18 / 4 = 4.5 cm."
    result: "(i) ವಿಸ್ತೀರ್ಣ 18 cm². (ii) ಪಾದ AD ಗೆ ಅನುರೂಪವಾದ ಎತ್ತರ 4.5 cm."

  [ವಿಶ್ಲೇಷಣೆ]
  ಸಮಾಂತರ ಚತುರ್ಭುಜದ ವಿಸ್ತೀರ್ಣದ ಪ್ರಮುಖ ಅಂಶವೆಂದರೆ 'ಎತ್ತರ'ವನ್ನು ಸರಿಯಾಗಿ ಗುರುತಿಸುವುದು. ಎತ್ತರವು ಪಕ್ಕದ ಬಾಹುವಿನ (slant height) ಉದ್ದವಲ್ಲ, ಬದಲಾಗಿ ಪಾದಕ್ಕೆ ಲಂಬವಾಗಿರುವ ದೂರ. ಪರೀಕ್ಷೆಗಳಲ್ಲಿ, ಒಂದು ಪಾದ ಮತ್ತು ಎತ್ತರವನ್ನು ನೀಡಿ ವಿಸ್ತೀರ್ಣವನ್ನು ಕೇಳಬಹುದು, ಅಥವಾ ವಿಸ್ತೀರ್ಣ ಮತ್ತು ಒಂದು ಪಾದವನ್ನು ನೀಡಿ ಇನ್ನೊಂದು ಪಾದದ ಅನುರೂಪ ಎತ್ತರವನ್ನು ಕೇಳಬಹುದು (ಮೇಲಿನ ಉದಾಹರಣೆಯಂತೆ).

  [ಸಂಬಂಧ]
  - ಪೂರ್ವಾಪೇಕ್ಷಿತ ಅವಧಾರಣೆಗಳು: ಆಯತದ ವಿಸ್ತೀರ್ಣ, ಲಂಬ ರೇಖೆಗಳು.
  - ಮುಂದಿನ ಅವಧಾರಣೆಗಳೊಂದಿಗೆ ಸಂಪರ್ಕ: ತ್ರಿಭುಜದ ವಿಸ್ತೀರ್ಣ (ತ್ರಿಭುಜವು ಸಮಾಂತರ ಚತುರ್ಭುಜದ ಅರ್ಧದಷ್ಟಿದೆ).

  [ಶೈಕ್ಷಣಿಕ_ದೃಷ್ಟಿಕೋನ]
  - gpstr_weightage: medium
  - MCQ ಸಾಧ್ಯತೆ: ಪಾದ, ಎತ್ತರ, ಮತ್ತು ಪಕ್ಕದ ಬಾಹುವಿನ ಉದ್ದವನ್ನು ನೀಡಿ, ವಿಸ್ತೀರ್ಣವನ್ನು ಕೇಳುವುದು (ಪಕ್ಕದ ಬಾಹುವನ್ನು ಬಳಸಿ ತಪ್ಪು ಉತ್ತರವನ್ನು ಆಯ್ಕೆ ಮಾಡುವಂತೆ ಪ್ರೇರೇಪಿಸುವುದು).
  - 2-ಅಂಕ ಪ್ರಶ್ನೆ: ನೇರ ಸೂತ್ರದ ಬಳಕೆ ಅಥವಾ ವಿಸ್ತೀರ್ಣದಿಂದ ಎತ್ತರವನ್ನು ಕಂಡುಹಿಡಿಯುವುದು.
  - 5-ಅಂಕ ಪ್ರಶ್ನೆ: ಇಲ್ಲ.
  - ಪ್ರಮುಖ trap: ಎತ್ತರ (height) ಮತ್ತು ಪಕ್ಕದ ಬಾಹುವಿನ ಉದ್ದ (slant height) ನಡುವಿನ ವ್ಯತ್ಯಾಸವನ್ನು ತಿಳಿಯದಿರುವುದು.

concept_id: MEN-AREA-1_concept_4
concept_name: ತ್ರಿಭುಜದ ವಿಸ್ತೀರ್ಣ (Area of a Triangle)
domain: Mensuration

  [ಸಾರಾಂಶ]
  ಯಾವುದೇ ತ್ರಿಭುಜದ ವಿಸ್ತೀರ್ಣವು, ಅದರ ಪಾದ ಮತ್ತು ಅನುರೂಪ ಎತ್ತರದ ಗುಣಲಬ್ಧದ ಅರ್ಧದಷ್ಟಿರುತ್ತದೆ. ಇದನ್ನು, ತ್ರಿಭುಜವು ಅದೇ ಪಾದ ಮತ್ತು ಎತ್ತರವನ್ನು ಹೊಂದಿರುವ ಸಮಾಂತರ ಚತುರ್ಭುಜದ ವಿಸ್ತೀರ್ಣದ ಅರ್ಧದಷ್ಟಿದೆ ಎಂದು ತೋರಿಸುವ ಮೂಲಕ ಸಾಧಿಸಬಹುದು.

  [ಸೂತ್ರಗಳು]
  - formula_name: "Area of a Triangle"
    latex: "A = \\frac{1}{2} \\times b \\times h"
    condition: "b = ತ್ರಿಭುಜದ ಪಾದ (base), h = ಪಾದಕ್ಕೆ ಎಳೆದ ಅನುರೂಪ ಎತ್ತರ (corresponding height)."
    used_in: "ಯಾವುದೇ ತ್ರಿಭುಜದ ವಿಸ್ತೀರ್ಣವನ್ನು ಲೆಕ್ಕಾಚಾರ ಮಾಡಲು."
    derivation:
      - step: 1
        action: "ಒಂದು ತ್ರಿಭುಜವನ್ನು ತೆಗೆದುಕೊಂಡು, ಅದಕ್ಕೆ ಸರ್ವಸಮವಾಗಿರುವ ಇನ್ನೊಂದು ತ್ರಿಭುಜವನ್ನು ರಚಿಸಿ."
        justification: "ಎರಡು ಸರ್ವಸಮ ತ್ರಿಭುಜಗಳನ್ನು ಬಳಸಿ ಹೊಸ ಆಕೃತಿ ರಚಿಸಲು."
      - step: 2
        action: "ಎರಡೂ ತ್ರಿಭುಜಗಳನ್ನು ಅವುಗಳ ಒಂದು ಅನುರೂಪ ಬಾಹುವಿನ ಉದ್ದಕ್ಕೂ ಜೋಡಿಸಿ."
        justification: "ಇದರಿಂದ ಒಂದು ಸಮಾಂತರ ಚತುರ್ಭುಜವು ಉಂಟಾಗುತ್ತದೆ."
      - step: 3
        action: "ಉಂಟಾದ ಸಮಾಂತರ ಚತುರ್ಭುಜದ ವಿಸ್ತೀರ್ಣವನ್ನು ಗಮನಿಸಿ."
        justification: "ಸಮಾಂತರ ಚತುರ್ಭುಜದ ವಿಸ್ತೀರ್ಣ = ಪಾದ × ಎತ್ತರ."
      - step: 4
        action: "ತ್ರಿಭುಜದ ವಿಸ್ತೀರ್ಣವನ್ನು ಸಮಾಂತರ ಚತುರ್ಭುಜದ ವಿಸ್ತೀರ್ಣಕ್ಕೆ ಸಂಬಂಧಿಸಿ."
        justification: "ಸಮಾಂತರ ಚತುರ್ಭುಜವು ಎರಡು ಸರ್ವಸಮ ತ್ರಿಭುಜಗಳಿಂದ ಮಾಡಲ್ಪಟ್ಟಿರುವುದರಿಂದ, ಪ್ರತಿ ತ್ರಿಭುಜದ ವಿಸ್ತೀರ್ಣವು ಸಮಾಂತರ ಚತುರ್ಭುಜದ ವಿಸ್ತೀರ್ಣದ ಅರ್ಧದಷ್ಟಿರುತ್ತದೆ."
        calculation: "ತ್ರಿಭುಜದ ವಿಸ್ತೀರ್ಣ = 1/2 × (ಸಮಾಂತರ ಚತುರ್ಭುಜದ ವಿಸ್ತೀರ್ಣ) = 1/2 × (b × h)."
    result: "ತ್ರಿಭುಜದ ವಿಸ್ತೀರ್ಣ = 1/2 × ಪಾದ × ಎತ್ತರ."

  [ಅಂಶಪಟ್ಟಿ / Properties]
  - ವಿಶಾಲಕೋನ ತ್ರಿಭುಜದಲ್ಲಿ (obtuse triangle), ಕೆಲವು ಶೃಂಗಗಳಿಂದ ಎಳೆದ ಎತ್ತರವು ತ್ರಿಭುಜದ ಹೊರಭಾಗದಲ್ಲಿರಬಹುದು.
  - ಸರ್ವಸಮ ತ್ರಿಭುಜಗಳು (congruent triangles) ಸಮಾನ ವಿಸ್ತೀರ್ಣವನ್ನು ಹೊಂದಿರುತ್ತವೆ.
  - ಆದರೆ, ಸಮಾನ ವಿಸ್ತೀರ್ಣವನ್ನು ಹೊಂದಿರುವ ತ್ರಿಭುಜಗಳು ಸರ್ವಸಮವಾಗಿರಬೇಕೆಂದೇನಿಲ್ಲ.

  [ಕ್ರಮಬದ್ಧ ಉದಾಹರಣೆ]
  - example_id: "MEN-AREA-1_example_5"
    difficulty: "intermediate"
    problem: "ΔPQR ನಲ್ಲಿ, PR = 8cm, QR = 4cm ಮತ್ತು PL (QR ಮೇಲಿನ ಎತ್ತರ) = 5cm ಆಗಿದೆ. (i) ΔPQRನ ವಿಸ್ತೀರ್ಣ ಮತ್ತು (ii) QM (PR ಮೇಲಿನ ಎತ್ತರ) ಗಳನ್ನು ಕಂಡುಹಿಡಿಯಿರಿ."
    steps:
      - step: 1
        action: "(i) ತ್ರಿಭುಜದ ವಿಸ್ತೀರ್ಣವನ್ನು ಲೆಕ್ಕಾಚಾರ ಮಾಡುವುದು."
        justification: "ಪಾದ QR ಮತ್ತು ಅದರ ಅನುರೂಪ ಎತ್ತರ PL ತಿಳಿದಿದೆ."
        calculation: "ವಿಸ್ತೀರ್ಣ = 1/2 × ಪಾದ × ಎತ್ತರ = 1/2 × QR × PL = 1/2 × 4 cm × 5 cm = 10 cm²."
      - step: 2
        action: "(ii) ಇನ್ನೊಂದು ಎತ್ತರ QM ಅನ್ನು ಕಂಡುಹಿಡಿಯುವುದು."
        justification: "ಈಗ PR ಅನ್ನು ಪಾದವಾಗಿ ಪರಿಗಣಿಸಿ. ವಿಸ್ತೀರ್ಣವು ಒಂದೇ ಆಗಿರುತ್ತದೆ."
        calculation: "ವಿಸ್ತೀರ್ಣ = 1/2 × PR × QM"
      - step: 3
        action: "ಸಮೀಕರಣವನ್ನು ಬಿಡಿಸುವುದು."
        justification: "ಹಂತ 1 ರಿಂದ ವಿಸ್ತೀರ್ಣದ ಮೌಲ್ಯವನ್ನು ಬಳಸಿ."
        calculation: "10 cm² = 1/2 × 8 cm × QM \n 10 = 4 × QM \n QM = 10 / 4 = 2.5 cm."
    result: "(i) ವಿಸ್ತೀರ್ಣ 10 cm². (ii) ಎತ್ತರ QM 2.5 cm."

  [ವಿಶ್ಲೇಷಣೆ]
  ತ್ರಿಭುಜದ ವಿಸ್ತೀರ್ಣವು ಮೆನ್ಸುರೇಶನ್‌ನ ಒಂದು ಮೂಲಭೂತ ಪರಿಕಲ್ಪನೆಯಾಗಿದ್ದು, ಅನೇಕ ಸಂಕೀರ್ಣ ಆಕೃತಿಗಳ ವಿಸ್ತೀರ್ಣವನ್ನು ಕಂಡುಹಿಡಿಯಲು ಬಳಸಲಾಗುತ್ತದೆ (ತ್ರಿಭುಜೀಕರಣದ ಮೂಲಕ). ಸಮಾಂತರ ಚತುರ್ಭುಜದಂತೆಯೇ, ಎತ್ತರವನ್ನು ಸರಿಯಾಗಿ ಗುರುತಿಸುವುದು ಮುಖ್ಯ.

  [ಸಂಬಂಧ]
  - ಪೂರ್ವಾಪೇಕ್ಷಿತ ಅವಧಾರಣೆಗಳು: ಸಮಾಂತರ ಚತುರ್ಭುಜದ ವಿಸ್ತೀರ್ಣ.
  - ಮುಂದಿನ ಅವಧಾರಣೆಗಳೊಂದಿಗೆ ಸಂಪರ್ಕ: ತ್ರಿಕೋನಮಿತಿ (Trigonometry), ಸಂಯುಕ್ತ ಆಕೃತಿಗಳ ವಿಸ್ತೀರ್ಣ.

  [ಶೈಕ್ಷಣಿಕ_ದೃಷ್ಟಿಕೋನ]
  - gpstr_weightage: high
  - MCQ ಸಾಧ್ಯತೆ: ಲಂಬಕೋನ ತ್ರಿಭುಜದ ವಿಸ್ತೀರ್ಣ, ಇಲ್ಲಿ ಲಂಬಬಾಹುಗಳೇ ಪಾದ ಮತ್ತು ಎತ್ತರವಾಗುತ್ತವೆ.
  - 2-ಅಂಕ ಪ್ರಶ್ನೆ: ವಿಸ್ತೀರ್ಣ ಮತ್ತು ಪಾದ ನೀಡಿ ಎತ್ತರ ಕಂಡುಹಿಡಿಯುವುದು, ಅಥವಾ ಮೇಲಿನ ಉದಾಹರಣೆಯಂತಹ ಲೆಕ್ಕ.
  - 5-ಅಂಕ ಪ್ರಶ್ನೆ: ಸಂಯುಕ್ತ ಆಕೃತಿಯಲ್ಲಿ ತ್ರಿಭುಜದ ವಿಸ್ತೀರ್ಣವನ್ನು ಬಳಸುವುದು.
  - ಪ್ರಮುಖ trap: ವಿಶಾಲಕೋನ ತ್ರಿಭುಜದ ಎತ್ತರವನ್ನು ಗುರುತಿಸುವಲ್ಲಿ ತಪ್ಪು ಮಾಡುವುದು. ಸೂತ್ರದಲ್ಲಿ 1/2 ಅನ್ನು ಮರೆಯುವುದು.

[ಮೂಲ_ಅವಧಾರಣೆ_5]
concept_id: MEN-AREA-1_concept_5
concept_name: ವೃತ್ತದ ಪರಿಧಿ (Circumference of a Circle)
domain: Mensuration

  [ಸಾರಾಂಶ]
  ವೃತ್ತದ ಸುತ್ತಲಿನ ದೂರವನ್ನು ಪರಿಧಿ ಎಂದು ಕರೆಯಲಾಗುತ್ತದೆ. ಯಾವುದೇ ವೃತ್ತದಲ್ಲಿ, ಅದರ ಪರಿಧಿ ಮತ್ತು ವ್ಯಾಸದ ನಡುವಿನ ಅನುಪಾತವು ಒಂದು ಸ್ಥಿರಾಂಕವಾಗಿರುತ್ತದೆ, ಇದನ್ನು ಪೈ ($\pi$) ಎಂದು ಕರೆಯಲಾಗುತ್ತದೆ. ಇದರ ಅಂದಾಜು ಮೌಲ್ಯ 22/7 ಅಥವಾ 3.14.

  [ವ್ಯಾಖ್ಯೆ]
  - **ಪರಿಧಿ (Circumference):** ವೃತ್ತಾಕಾರದ ವಲಯದ ಸುತ್ತಲಿನ ದೂರವನ್ನು ಪರಿಧಿ ಎನ್ನುತ್ತಾರೆ.
  - **ಪೈ (Pi - $\pi$):** ವೃತ್ತದ ಪರಿಧಿ (C) ಮತ್ತು ಅದರ ವ್ಯಾಸ (d) ನಡುವಿನ ಅನುಪಾತ. $\pi = C/d$.

  [ಸೂತ್ರಗಳು]
  - formula_name: "Circumference of a Circle (using diameter)"
    latex: "C = \\pi d"
    condition: "d = ವೃತ್ತದ ವ್ಯಾಸ (diameter)."
    used_in: "ವ್ಯಾಸ ತಿಳಿದಿರುವಾಗ ಪರಿಧಿಯನ್ನು ಕಂಡುಹಿಡಿಯಲು."

  - formula_name: "Circumference of a Circle (using radius)"
    latex: "C = 2\\pi r"
    condition: "r = ವೃತ್ತದ ತ್ರಿಜ್ಯ (radius)."
    derived_from: "C = πd ಮತ್ತು d = 2r"
    used_in: "ತ್ರಿಜ್ಯ ತಿಳಿದಿರುವಾಗ ಪರಿಧಿಯನ್ನು ಕಂಡುಹಿಡಿಯಲು."

  - formula_name: "Perimeter of a Semicircle"
    latex: "P = \\pi r + 2r"
    condition: "r = ಅರ್ಧವೃತ್ತದ ತ್ರಿಜ್ಯ (radius)."
    derived_from: "Perimeter = (Half of circumference) + diameter = (1/2 * 2πr) + 2r"
    used_in: "ಅರ್ಧವೃತ್ತಾಕಾರದ ವಸ್ತುಗಳ ಸುತ್ತಳತೆ ಕಂಡುಹಿಡಿಯಲು."

  [ಕ್ರಮಬದ್ಧ ಉದಾಹರಣೆ]
  - example_id: "MEN-AREA-1_example_6"
    difficulty: "intermediate"
    problem: "ಸುಧಾಂಶು 7 cm ತ್ರಿಜ್ಯವುಳ್ಳ ವೃತ್ತಾಕಾರದ ಡಿಸ್ಕ್ ಅನ್ನು ಎರಡು ಸಮಭಾಗಗಳಾಗಿ ವಿಭಾಗಿಸುತ್ತಾಳೆ. ಪ್ರತಿ ಅರ್ಧವೃತ್ತಾಕಾರದ ಡಿಸ್ಕ್‌ನ ಸುತ್ತಳತೆ ಎಷ್ಟು? ($\pi = 22/7$ ಎಂದು ಬಳಸಿ)."
    steps:
      - step: 1
        action: "ಸುತ್ತಳತೆಯ ಭಾಗಗಳನ್ನು ಗುರುತಿಸುವುದು."
        justification: "ಅರ್ಧವೃತ್ತದ ಸುತ್ತಳತೆಯು ವೃತ್ತದ ಪರಿಧಿಯ ಅರ್ಧ ಭಾಗ ಮತ್ತು ವ್ಯಾಸವನ್ನು ಒಳಗೊಂಡಿರುತ್ತದೆ."
        calculation: "ಸುತ್ತಳತೆ = ಅರ್ಧವೃತ್ತದ ಕಂಸದ ಉದ್ದ + ವ್ಯಾಸ."
      - step: 2
        action: "ಅರ್ಧವೃತ್ತದ ಕಂಸದ ಉದ್ದವನ್ನು ಲೆಕ್ಕಾಚಾರ ಮಾಡುವುದು."
        justification: "ಇದು ಪೂರ್ಣ ವೃತ್ತದ ಪರಿಧಿಯ ಅರ್ಧದಷ್ಟಿದೆ."
        calculation: "ಕಂಸದ ಉದ್ದ = (1/2) × 2πr = πr = (22/7) × 7 cm = 22 cm."
      - step: 3
        action: "ವ್ಯಾಸವನ್ನು ಲೆಕ್ಕಾಚಾರ ಮಾಡುವುದು."
        justification: "ವ್ಯಾಸವು ತ್ರಿಜ್ಯದ ಎರಡರಷ್ಟಿರುತ್ತದೆ."
        calculation: "ವ್ಯಾಸ (d) = 2r = 2 × 7 cm = 14 cm."
      - step: 4
        action: "ಒಟ್ಟು ಸುತ್ತಳತೆಯನ್ನು ಕಂಡುಹಿಡಿಯುವುದು."
        justification: "ಹಂತ 2 ಮತ್ತು 3 ರ ಮೌಲ್ಯಗಳನ್ನು ಕೂಡಿಸುವುದು."
        calculation: "ಒಟ್ಟು ಸುತ್ತಳತೆ = 22 cm + 14 cm = 36 cm."
    result: "ಪ್ರತಿ ಅರ್ಧವೃತ್ತಾಕಾರದ ಡಿಸ್ಕ್‌ನ ಸುತ್ತಳತೆ 36 cm."

  [ವಿಶ್ಲೇಷಣೆ]
  ವೃತ್ತದ ಪರಿಧಿಯ ಲೆಕ್ಕಾಚಾರಗಳು ಚಕ್ರಗಳು, ಪಥಗಳು ಮತ್ತು ವೃತ್ತಾಕಾರದ ವಸ್ತುಗಳಿಗೆ ಸಂಬಂಧಿಸಿದ ಅನ್ವಯಿಕ ಸಮಸ್ಯೆಗಳಲ್ಲಿ ಮುಖ್ಯವಾಗಿವೆ. $\pi$ ನ ಯಾವ ಮೌಲ್ಯವನ್ನು (22/7 ಅಥವಾ 3.14) ಬಳಸಬೇಕೆಂದು ಪ್ರಶ್ನೆಯಲ್ಲಿ ಸೂಚಿಸಲಾಗಿದೆಯೇ ಎಂದು ಗಮನಿಸುವುದು ಮುಖ್ಯ. ತ್ರಿಜ್ಯ ಅಥವಾ ವ್ಯಾಸವು 7ರ ಅಪವರ್ತ್ಯವಾಗಿದ್ದರೆ, 22/7 ಬಳಸುವುದು ಲೆಕ್ಕಾಚಾರವನ್ನು ಸುಲಭಗೊಳಿಸುತ್ತದೆ.

  [ಸಂಬಂಧ]
  - ಪೂರ್ವಾಪೇಕ್ಷಿತ ಅವಧಾರಣೆಗಳು: ತ್ರಿಜ್ಯ, ವ್ಯಾಸ, ಅನುಪಾತ.
  - ಮುಂದಿನ ಅವಧಾರಣೆಗಳೊಂದಿಗೆ ಸಂಪರ್ಕ: ವೃತ್ತದ ವಿಸ್ತೀರ್ಣ, ಕಂಸದ ಉದ್ದ, ತ್ರಿಜ್ಯಾಂತರ ಖಂಡದ ಸುತ್ತಳತೆ.

  [ಶೈಕ್ಷಣಿಕ_ದೃಷ್ಟಿಕೋನ]
  - gpstr_weightage: high
  - MCQ ಸಾಧ್ಯತೆ: ತ್ರಿಜ್ಯ/ವ್ಯಾಸ ನೀಡಿ ಪರಿಧಿ ಕೇಳುವುದು. ಚಕ್ರವು ನಿರ್ದಿಷ್ಟ ದೂರವನ್ನು ಕ್ರಮಿಸಲು ಎಷ್ಟು ಸುತ್ತು ಸುತ್ತಬೇಕು ಎಂಬ ಪ್ರಶ್ನೆಗಳು.
  - 2-ಅಂಕ ಪ್ರಶ್ನೆ: ಅರ್ಧವೃತ್ತದ ಸುತ್ತಳತೆ ಅಥವಾ ಪರಿಧಿ ಮತ್ತು ತ್ರಿಜ್ಯದ ನಡುವಿನ ಸಂಬಂಧದ ಸರಳ ಲೆಕ್ಕಗಳು.
  - 5-ಅಂಕ ಪ್ರಶ್ನೆ: ಓಟದ ಪಥದಂತಹ ಸಂಯುಕ್ತ ಆಕೃತಿಗಳ ಸುತ್ತಳತೆ.
  - ಪ್ರಮುಖ trap: ಅರ್ಧವೃತ್ತದ ಸುತ್ತಳತೆಯನ್ನು ಲೆಕ್ಕಾಚಾರ ಮಾಡುವಾಗ ವ್ಯಾಸವನ್ನು ಸೇರಿಸಲು ಮರೆಯುವುದು (ಕೇವಲ πr ಎಂದು ಲೆಕ್ಕ ಹಾಕುವುದು).

[ಮೂಲ_ಅವಧಾರಣೆ_6]
concept_id: MEN-AREA-1_concept_6
concept_name: ವೃತ್ತದ ವಿಸ್ತೀರ್ಣ (Area of a Circle)
domain: Mensuration

  [ಸಾರಾಂಶ]
  ವೃತ್ತವು ಆವರಿಸುವ ಪ್ರದೇಶವನ್ನು ಅದರ ವಿಸ್ತೀರ್ಣ ಎನ್ನುತ್ತಾರೆ. ಇದನ್ನು ವೃತ್ತದ ತ್ರಿಜ್ಯದ ವರ್ಗ ಮತ್ತು $\pi$ ನ ಗುಣಲಬ್ಧವಾಗಿ ಲೆಕ್ಕಹಾಕಲಾಗುತ್ತದೆ. ವೃತ್ತವನ್ನು ಅನೇಕ ತ್ರಿಜ್ಯಾಂತರ ಖಂಡಗಳಾಗಿ ವಿಭಜಿಸಿ, ಅವುಗಳನ್ನು ಒಂದು ಆಯತದ ರೂಪದಲ್ಲಿ ಜೋಡಿಸುವ ಮೂಲಕ ಈ ಸೂತ್ರವನ್ನು ದೃಶ್ಯೀಕರಿಸಬಹುದು.

  [ಸೂತ್ರಗಳು]
  - formula_name: "Area of a Circle"
    latex: "A = \\pi r^2"
    condition: "r = ವೃತ್ತದ ತ್ರಿಜ್ಯ (radius)."
    used_in: "ವೃತ್ತಾಕಾರದ ಪ್ರದೇಶಗಳ ವಿಸ್ತೀರ್ಣವನ್ನು ಲೆಕ್ಕಾಚಾರ ಮಾಡಲು."
    derivation:
      - step: 1
        action: "ವೃತ್ತವನ್ನು ಅನೇಕ (ಉದಾ. 64) ಸಮಾನ ತ್ರಿಜ್ಯಾಂತರ ಖಂಡಗಳಾಗಿ (sectors) ವಿಭಜಿಸಿ."
        justification: "ಆಕೃತಿಯನ್ನು ಮರುಜೋಡಣೆಗಾಗಿ ಸಣ್ಣ ಭಾಗಗಳಾಗಿ ಕತ್ತರಿಸುವುದು."
      - step: 2
        action: "ಈ ತ್ರಿಜ್ಯಾಂತರ ಖಂಡಗಳನ್ನು ಪಕ್ಕಪಕ್ಕದಲ್ಲಿ, ಒಂದರ ಪಕ್ಕ ಇನ್ನೊಂದನ್ನು ತಲೆಕೆಳಗಾಗಿ ಜೋಡಿಸಿ."
        justification: "ಇದರಿಂದ ಒಂದು ಸರಿಸುಮಾರು ಆಯತದ ಆಕಾರ ಉಂಟಾಗುತ್ತದೆ."
      - step: 3
        action: "ಉಂಟಾದ ಆಯತದ ಉದ್ದ ಮತ್ತು ಅಗಲವನ್ನು ಗುರುತಿಸಿ."
        justification: "ಆಯತದ ಅಗಲವು ವೃತ್ತದ ತ್ರಿಜ್ಯಕ್ಕೆ (r) ಸಮನಾಗಿರುತ್ತದೆ. ಆಯತದ ಉದ್ದವು ವೃತ್ತದ ಪರಿಧಿಯ ಅರ್ಧದಷ್ಟಿರುತ್ತದೆ (πr), ಏಕೆಂದರೆ ತ್ರಿಜ್ಯಾಂತರ ಖಂಡಗಳ ಕಂಸಗಳು ಆಯತದ ಮೇಲಿನ ಮತ್ತು ಕೆಳಗಿನ ಬದಿಗಳನ್ನು ರೂಪಿಸುತ್ತವೆ."
        calculation: "ಉದ್ದ = 1/2 × C = 1/2 × 2πr = πr. ಅಗಲ = r."
      - step: 4
        action: "ಆಯತದ ವಿಸ್ತೀರ್ಣವನ್ನು ಲೆಕ್ಕಾಚಾರ ಮಾಡಿ."
        justification: "ವಿಸ್ತೀರ್ಣ = ಉದ್ದ × ಅಗಲ."
        calculation: "ವಿಸ್ತೀರ್ಣ = (πr) × r = πr²."
    result: "ವೃತ್ತದ ವಿಸ್ತೀರ್ಣ = πr²."

  - formula_name: "Area of a Ring (Annulus)"
    latex: "A = \\pi (R^2 - r^2)"
    condition: "R = ಹೊರ ವೃತ್ತದ ತ್ರಿಜ್ಯ (Outer radius), r = ಒಳ ವೃತ್ತದ ತ್ರಿಜ್ಯ (Inner radius)."
    derived_from: "Area of Ring = (Area of Outer Circle) - (Area of Inner Circle) = πR² - πr²"
    used_in: "ವೃತ್ತಾಕಾರದ ಪಥ, ಕಂಕಣ ಮುಂತಾದವುಗಳ ವಿಸ್ತೀರ್ಣವನ್ನು ಕಂಡುಹಿಡಿಯಲು."

  [ಕ್ರಮಬದ್ಧ ಉದಾಹರಣೆ]
  - example_id: "MEN-AREA-1_example_7"
    difficulty: "intermediate"
    problem: "ಒಂದೇ ಕೇಂದ್ರವಿರುವ ಎರಡು ವೃತ್ತಗಳಿವೆ. ದೊಡ್ಡ ವೃತ್ತದ ತ್ರಿಜ್ಯ 10 cm ಮತ್ತು ಚಿಕ್ಕ ವೃತ್ತದ ತ್ರಿಜ್ಯ 4 cm. ಎರಡು ವೃತ್ತಗಳ ನಡುವಿನ ಛಾಯೀಕೃತ ಭಾಗದ (ಕಂಕಣ) ವಿಸ್ತೀರ್ಣವನ್ನು ಕಂಡುಹಿಡಿಯಿರಿ. ($\pi = 3.14$ ಬಳಸಿ)."
    steps:
      - step: 1
        action: "ದೊಡ್ಡ ವೃತ್ತದ ವಿಸ್ತೀರ್ಣವನ್ನು ಲೆಕ್ಕಾಚಾರ ಮಾಡುವುದು."
        justification: "A = πR² ಸೂತ್ರವನ್ನು ಬಳಸಿ."
        calculation: "ದೊಡ್ಡ ವೃತ್ತದ ವಿಸ್ತೀರ್ಣ = 3.14 × (10 cm)² = 3.14 × 100 = 314 cm²."
      - step: 2
        action: "ಚಿಕ್ಕ ವೃತ್ತದ ವಿಸ್ತೀರ್ಣವನ್ನು ಲೆಕ್ಕಾಚಾರ ಮಾಡುವುದು."
        justification: "A = πr² ಸೂತ್ರವನ್ನು ಬಳಸಿ."
        calculation: "ಚಿಕ್ಕ ವೃತ್ತದ ವಿಸ್ತೀರ್ಣ = 3.14 × (4 cm)² = 3.14 × 16 = 50.24 cm²."
      - step: 3
        action: "ಛಾಯೀಕೃತ ಭಾಗದ ವಿಸ್ತೀರ್ಣವನ್ನು ಕಂಡುಹಿಡಿಯುವುದು."
        justification: "ಇದು ದೊಡ್ಡ ವೃತ್ತದ ವಿಸ್ತೀರ್ಣ ಮತ್ತು ಚಿಕ್ಕ ವೃತ್ತದ ವಿಸ್ತೀರ್ಣದ ನಡುವಿನ ವ್ಯತ್ಯಾಸ."
        calculation: "ಕಂಕಣದ ವಿಸ್ತೀರ್ಣ = 314 cm² - 50.24 cm² = 263.76 cm²."
    result: "ಛಾಯೀಕೃತ ಭಾಗದ ವಿಸ್ತೀರ್ಣ 263.76 cm²."

  [ವಿಶ್ಲೇಷಣೆ]
  ವೃತ್ತದ ವಿಸ್ತೀರ್ಣವು ವೃತ್ತಾಕಾರದ ಜಾಗದ ಅಳತೆಗೆ ಸಂಬಂಧಿಸಿದ ಸಮಸ್ಯೆಗಳಲ್ಲಿ (ಉದಾ. ವೃತ್ತಾಕಾರದ ಉದ್ಯಾನ, ಪಿಜ್ಜಾ, ಇತ್ಯಾದಿ) ಬಳಸಲ್ಪಡುತ್ತದೆ. ಪರಿಧಿಯಂತೆಯೇ, $\pi$ ನ ಸರಿಯಾದ ಮೌಲ್ಯವನ್ನು ಬಳಸುವುದು ಮುಖ್ಯ. ವಿಸ್ತೀರ್ಣದ ಲೆಕ್ಕಾಚಾರದಲ್ಲಿ ತ್ರಿಜ್ಯವನ್ನು ವರ್ಗೀಕರಿಸಲಾಗುತ್ತದೆ ($r^2$), ಇದನ್ನು ಮರೆಯಬಾರದು.

  [ಸಂಬಂಧ]
  - ಪೂರ್ವಾಪೇಕ್ಷಿತ ಅವಧಾರಣೆಗಳು: ವೃತ್ತದ ಪರಿಧಿ, ಘಾತಗಳು (exponents).
  - ಮುಂದಿನ ಅವಧಾರಣೆಗಳೊಂದಿಗೆ ಸಂಪರ್ಕ: ಸಿಲಿಂಡರ್ ಮತ್ತು ಶಂಕುವಿನ ಘನಫಲ ಮತ್ತು ಮೇಲ್ಮೈ ವಿಸ್ತೀರ್ಣ.

  [ಶೈಕ್ಷಣಿಕ_ದೃಷ್ಟಿಕೋನ]
  - gpstr_weightage: high
  - MCQ ಸಾಧ್ಯತೆ: ವಿಸ್ತೀರ್ಣ ನೀಡಿ ತ್ರಿಜ್ಯ ಅಥವಾ ಪರಿಧಿಯನ್ನು ಕಂಡುಹಿಡಿಯುವುದು.
  - 2-ಅಂಕ ಪ್ರಶ್ನೆ: ಕಂಕಣದ ವಿಸ್ತೀರ್ಣ (Area of ring) ಕಂಡುಹಿಡಿಯುವುದು.
  - 5-ಅಂಕ ಪ್ರಶ್ನೆ: ವೃತ್ತಾಕಾರದ ಪಥವನ್ನು ನಿರ್ಮಿಸಲು ತಗಲುವ ವೆಚ್ಚ ಅಥವಾ ಸಂಯುಕ್ತ ಆಕೃತಿಗಳಲ್ಲಿ ವೃತ್ತದ ವಿಸ್ತೀರ್ಣವನ್ನು ಬಳಸುವುದು.
  - ಪ್ರಮುಖ trap: $A = \pi r^2$ ಬದಲು $A = 2\pi r$ (ಪರಿಧಿಯ ಸೂತ್ರ) ಬಳಸುವುದು. ತ್ರಿಜ್ಯದ ಬದಲು ವ್ಯಾಸವನ್ನು ಬಳಸಿ ಲೆಕ್ಕ ಮಾಡುವುದು.
