AVAILABLE CSS CLASSES — use ONLY these

PAGE STRUCTURE:
  .page-header          — centered header block (topic title, domain, weightage)
  .page-header h1       — topic name
  .domain-tag           — domain label inside header
  .weightage-badge      — with .weightage-high / .weightage-medium / .weightage-low

TABLE OF CONTENTS:
  .toc                  — TOC container
  .toc h2               — TOC heading
  .toc ol / .toc li     — numbered concept links
  .toc a                — clickable link

FORMULA QUICK REFERENCE:
  .formula-grid         — grid container for formula cards
  .formula-card         — individual formula card
  .formula-card .fname  — formula name (small text)
  .formula-card .flatex — formula LaTeX (large, centered)

CONCEPT SECTIONS:
  .concept              — main concept container (one per ಮೂಲ_ಅವಧಾರಣೆ)
  .concept-header       — header area within concept
  .concept-number       — small uppercase label ("ಅವಧಾರಣೆ 1")
  .concept-title        — concept name h2
  .summary              — summary paragraph
  .prereq-tags          — prerequisite tag container
  .prereq-tag           — individual prerequisite tag
  .section-title        — section heading within concept (h3)

DEFINITIONS:
  .definition-box       — blue left-border card
  .definition-box .term — bold term name
  .definition-box .def-text — definition text

FORMULAS:
  .formula-box          — green left-border card
  .formula-box .formula-name    — formula name (green, small)
  .formula-box .formula-latex   — large centered LaTeX
  .formula-box .formula-meta    — condition/exception text
  details.derivation            — collapsible derivation
  details.derivation summary    — "ವ್ಯುತ್ಪತ್ತಿ ▸" toggle
  .deriv-step                   — each derivation step

THEOREMS:
  .theorem-box          — purple left-border card
  .theorem-box .theorem-name      — theorem name
  .theorem-box .theorem-statement — italic quote-style statement
  details.proof                   — collapsible proof
  details.proof summary           — "ಸಾಬೀತು ▸" toggle

PROPERTIES:
  .property-list        — ol/ul for properties
  .property-list li     — individual property

WORKED EXAMPLES:
  .example              — example container
  .example-header       — header with difficulty badge
  .difficulty-badge     — with .diff-basic / .diff-intermediate / .diff-advanced
  .problem-box          — highlighted problem statement
  .step                 — one solution step (flex row)
  .step-number          — circular step number
  .step-content         — step body
  .step-action          — action text
  .step-calc            — calculation (LaTeX)
  .step-justify         — justification (small, muted)
  .answer-box           — green success box for final answer
  .trap-warning         — orange warning box for common traps

EXAM TIPS:
  .exam-tip             — exam intelligence card
  .exam-tip h4          — heading
  .exam-tip .tip-row    — each tip line

DIAGRAM PLACEHOLDER:
  .diagram-placeholder  — dashed border container
  .diagram-label        — "📐 ಚಿತ್ರ ಅಗತ್ಯ" label

FOOTER:
  .page-footer          — centered footer with stats
